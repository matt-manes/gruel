 60    def get_bases(self, object: Any) -> list[Any]:
 61        """Returns a recursive list of all the classes `object` inherits from."""
 62        parents: list[Any] = []
 63        bases = object.__bases__

 55class Gruel:
 56    """Scraper base class.
 57
 58    Classes subclassing `Gruel` need to implement the following methods:
 59
 60    * `get_parsable_items(self) -> list[Any]`
 61    * `parse_item(self, item: Any)->Any`
 62    * `store_item(self, item: Any)`
 63
 64    Calling the `scrape()` method will execute:
 65    1. `self.prescrape_chores()` (does nothing unless overridden)
 66    2. `self.get_parsable_items()`
 67    3. `self.parse_item()` for each item returned by `self.get_parsable_items()`
 68    4. `self.store_item()` for each successfully parsed item
 69    5. `self.postscrape_chores()` (only closes this instance's log file unless overridden)
 70
 71    When overriding `self.postscrape_chores`, it's recommended to either
 72    call `super().postscrape_chores()` or make sure to call `self.log.close()`.
 73    Otherwise running a large number of scrapers can cause file handle limit issues."""
 74
 75    def __init__(self, name: str | None = None, log_dir: Pathish | None = None):
 76        """
 77        :params:
 78        * `name`: The name of this scraper. If `None`, the name will be the stem of the file this class/subclass was defined in.
 79        i.e. A `Gruel` subclass located in a file called `myscraper.py` will have the name `"myscraper"`.
 80        * `log_dir`: The directory this scraper's logs should be saved to.
 81        If `None`, the logs will be written to a folder called `"gruel_logs"` within the current working directory.
 82        """
 83        self._name = name
 84        self._init_logger(log_dir)
 85        self.timer = Timer()
 86        self.success_count = 0
 87        self.fail_count = 0
 88        self.failed_to_get_parsable_items = False
 89        self.unexpected_failure_occured = False
 90        self.parsable_items: list[ParsableItem] = []
 91        self.parsed_items: list[Any] = []
 92
 93    @property
 94    def name(self) -> str:
 95        """Returns the name given to __init__ or the stem of the file this instance was defined in if one wasn't given."""
 96        return self._name or Pathier(inspect.getsourcefile(type(self))).stem  # type: ignore
 97
 98    @property
 99    def had_failures(self) -> bool:
100        """`True` if getting parsable items, parsing items, or unexpected failures occured."""
101        return (

Scraper base class.

Classes subclassing Gruel need to implement the following methods:

  • get_parsable_items(self) -> list[Any]
  • parse_item(self, item: Any)->Any
  • store_item(self, item: Any)

Calling the scrape() method will execute:

  1. self.prescrape_chores() (does nothing unless overridden)
  2. self.get_parsable_items()
  3. self.parse_item() for each item returned by self.get_parsable_items()
  4. self.store_item() for each successfully parsed item
  5. self.postscrape_chores() (only closes this instance's log file unless overridden)

When overriding self.postscrape_chores, it's recommended to either call super().postscrape_chores() or make sure to call self.log.close(). Otherwise running a large number of scrapers can cause file handle limit issues.
